## DSL

```yml
# 范围查询
{
	"query": {
		"range": {
			"id": {
				"gte": 0,
				"lte": 100
			}
		}
	}
}

# 前缀查询
{
	"query": {
		"prefix": {
			"author": {
				"value": "孟"
			}
		}
	}
}

# 通配符查询 * ？
{
	"query": {
		"wildcard": {
			"title": {
				"value": "过*"
			}
		}
	}
}
# ids 查询
{
	"query": {
		 "ids": {
		    "values":[1]
		 }
	}
}

```
